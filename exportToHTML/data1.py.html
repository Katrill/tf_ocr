import PIL
import os
from PIL import Image, ImageFont, ImageDraw


def get_image(str):
    # create an image
    im = PIL.Image.new ("1", size=(28, 28), color=1)
    # get a font
    font = ImageFont.truetype ("arial.ttf", 25)
    # get a drawing context
    d = ImageDraw.Draw (im)
    # draw multiline text
    d.text((14, 14), str, font=font,
            anchor='mm',
            fill=0)
    # im.show()
    par_dir = r"C:\Users\Admin\Downloads\alphabet"
    new_dir = f"{str.lower()}"
    path = os.path.join(par_dir, new_dir)
    if not os.path.exists(path):
        os.makedirs(path)
        im.save(path + "\\" + f"{str.lower()}" + "1" + ".png")
    else:
        im.save(path + "\\" + f"{str.lower ()}" + "2" + ".png")

for letter in letters[:len(letters):22]:
